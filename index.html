#04 · Contador de vocales avanzado conteo individual | estadísticas | vocal más repetida
📌 Pide una palabra, cuenta total de vocales, cuántas veces aparece cada una (solo muestra las que tienen al menos 1), y determina la(s) vocal(es) que más se repiten. 
// ============================================================
// PROGRAMA: Contador de vocales avanzado
// ============================================================
#include <iostream>
using namespace std;

int main() {
    string palabra;
    int vocales = 0;
    int vocalesA = 0, vocalesE = 0, vocalesI = 0, vocalesO = 0, vocalesU = 0;

    cout << "Ingresa una palabra: ";
    cin >> palabra;

    for(int i = 0; i < palabra.length(); i++) {
        // Contar total de vocales
        if(palabra[i] == 'a' || palabra[i] == 'e' ||
           palabra[i] == 'i' || palabra[i] == 'o' ||
           palabra[i] == 'u' || palabra[i] == 'A' ||
           palabra[i] == 'E' || palabra[i] == 'I' ||
           palabra[i] == 'O' || palabra[i] == 'U') {
            vocales++;
        }
        // Contar cada vocal individualmente
        if(palabra[i] == 'a' || palabra[i] == 'A') vocalesA++;
        else if(palabra[i] == 'e' || palabra[i] == 'E') vocalesE++;
        else if(palabra[i] == 'i' || palabra[i] == 'I') vocalesI++;
        else if(palabra[i] == 'o' || palabra[i] == 'O') vocalesO++;
        else if(palabra[i] == 'u' || palabra[i] == 'U') vocalesU++;
    }

    cout << "La palabra " << palabra << " tiene " << vocales << " vocales.\n\n";
    cout << "RESUMEN DE LAS VOCALES\n";
    if(vocalesA > 0) cout << "a = " << vocalesA << " veces.\n";
    if(vocalesE > 0) cout << "e = " << vocalesE << " veces.\n";
    if(vocalesI > 0) cout << "i = " << vocalesI << " veces.\n";
    if(vocalesO > 0) cout << "o = " << vocalesO << " veces.\n";
    if(vocalesU > 0) cout << "u = " << vocalesU << " veces.\n";

    // Determinar la(s) vocal(es) más repetida(s)
    int peso = vocalesA;
    if(vocalesE > peso) peso = vocalesE;
    if(vocalesI > peso) peso = vocalesI;
    if(vocalesO > peso) peso = vocalesO;
    if(vocalesU > peso) peso = vocalesU;

    cout << "La(s) vocal(es) mas repetida(s): ";
    if(vocalesA == peso) cout << "a ";
    if(vocalesE == peso) cout << "e ";
    if(vocalesI == peso) cout << "i ";
    if(vocalesO == peso) cout << "o ";
    if(vocalesU == peso) cout << "u ";
    cout << "con " << peso << " apariciones." << endl;
    return 0;
}
#03 · Número mayor (comparaciones con if) 4 números | if secuenciales

/*8. Número mayor
   Pide 4 números e indica cuál es el mayor.*/

#include <iostream>
using namespace std;

int main(){
    int a, b, c, d;
    
    cout << "Ingresa cuatro números: " << endl;
    cin >> a >> b >> c >> d;
    
    int mano = a;   // suponemos que 'a' es el mayor por ahora
    
    // Comparaciones sucesivas (nota: solo comparan con 'a', no con el valor actual de 'mano')
    if (b > a) {
        mano = b;
    }
    if (c > a) {
        mano = c;
    }
    if (d > a) {
        mano = d;
    }
    
    // Mostrar el resultado según qué variable quedó en 'mano'
    if (mano == a) {
        cout << "El número mayor es: " << a << endl;
    }
    if (mano == b) {
        cout << "El número mayor es: " << b << endl;
    }
    if (mano == c) {
        cout << "El número mayor es: " << c << endl;
    }
    if (mano == d) {
        cout << "El número mayor es: " << d << endl;
    }
    
    return 0;
}
