#01 · Opers. con 3 enteros

Lee tres enteros, muestra suma, diferencia, producto y cociente entero consecutivo. 
#include <iostream>
using namespace std;

int main()
{
    int a,b,c;

    cout<<"Dame un entero:"<<endl;
    cin>>a;

    cout<<"Dame otro entero:"<<endl;
    cin>>b;

    cout<<"Dame otro entero:"<<endl;
    cin>>c;

    cout<<"RESULTADOS"<<endl;
    cout<<"Suma: "<<a+b+c<<endl;
    cout<<"Diferencia: "<<a-b-c<<endl;
    cout<<"Producto: "<<a*b*c<<endl;
    cout<<"Cociente: "<<a/b/c<<endl;

    return 0;
}

#02 · Contador de vocales

Ingresa una palabra y cuenta cuántas vocales tiene (a, e, i, o, u). 
/* 2. Contar vocales en una palabra

Pide una palabra (string) y cuenta cuántas vocales tiene.

Ejemplo:

Entrada:
computadora

Salida:
Cantidad de vocales: 5*/

#include <iostream> 
using namespace std;

int main() {
    //DECLARACION DE VARIABLES //tipos??? = STRING 
    //tipo de variable = string la variable es palabra
    string palabra;//sirve para almacenar el valor dado por el usuario
    //tipo de variable = int(entero) la variable contador
    int contador = 0;//para contar las vocales
    
    cout<<"Ingresa la palabra: ";//muestra mensaje en la pantalla
    cin>>palabra;//almacena la palabra dada en la variable palabra
    
    //LOGICA DE DATOS //que es lo que haremos con lo que da el usuario??
    //ciclo for desde 0 hasta el  tamaño de palabra de uno en uno
    //iterar sobre la palabra dada y comparar letra * la poscion actual
    for(int i = 0; i< palabra.length(); i++){
        //CONDICION de comparacion posicion actual con vocales 
        if(palabra[i] == 'a'||palabra[i] == 'e'||
           palabra[i] == 'i'||palabra[i] == 'o'||
           palabra[i] == 'u')
        {
           contador++;//se suma el contador si hay vocal
        }//termina condicion
    }
    
    //RESULTADO
    //enseñar la salida de la vocales con la variable que la almaceno
    cout<<"La palabra "<<palabra<<endl;
    cout<<"Vocales que tiene es:"<<contador;

    return 0;
}
