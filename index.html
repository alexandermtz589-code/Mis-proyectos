#02 · Contador de vocales (básico) strings | mayúsculas

using namespace std;

{
    string palabra;
    int vocales = 0;

#03 · Mayor y menor de 4 números array | bucle | comparaciones
📌 Pide cuatro números enteros, los guarda en un array y determina cuál es el mayor y cuál es el menor mediante un bucle `for`. 
// ============================================================
// PROGRAMA: Número mayor y menor de 4 números
// ============================================================
#include <iostream>
using namespace std;

int main() {
    int a, b, c, d;

    cout << "Ingresa cuatro números enteros separados por espacio: ";
    cin >> a >> b >> c >> d;

    int numeros[4] = {a, b, c, d};
    int mayor = numeros[0];
    int menor = numeros[0];

    for(int i = 1; i < 4; i++) {
        if(numeros[i] > mayor) {
            mayor = numeros[i];
        }
        if(numeros[i] < menor) {
            menor = numeros[i];
        }
    }

    cout << "El número mayor es: " << mayor << endl;
    cout << "El número menor es: " << menor << endl;

    return 0;
}
#04 · Contador de vocales avanzado conteo individual | estadísticas | vocal más repetida
📌 Pide una palabra, cuenta total de vocales, cuántas veces aparece cada una (solo muestra las que tienen al menos 1), y determina la(s) vocal(es) que más se repiten. Totalmente documentado. 
// ============================================================
// PROGRAMA: Contador de vocales avanzado
// ============================================================
#include <iostream>
using namespace std;

int main() {
    // -------------------- DECLARACIÓN DE VARIABLES --------------------
    string palabra;      // Almacena la palabra ingresada por el usuario
    int vocales = 0;     // Contador total de vocales
    int vocalesA = 0;    // Contador específico para 'a' y 'A'
    int vocalesE = 0;    // Contador específico para 'e' y 'E'
    int vocalesI = 0;    // Contador específico para 'i' y 'I'
    int vocalesO = 0;    // Contador específico para 'o' y 'O'
    int vocalesU = 0;    // Contador específico para 'u' y 'U'

    // -------------------- ENTRADA DE DATOS --------------------
    cout << "Ingresa una palabra: ";
    cin >> palabra;     // Se lee una sola palabra (sin espacios)

    // -------------------- RECORRIDO CARÁCTER POR CARÁCTER --------------------
    for(int i = 0; i < palabra.length(); i++) {
        // Contar vocales en general (mayúsculas y minúsculas)
        if(palabra[i] == 'a' || palabra[i] == 'e' ||
           palabra[i] == 'i' || palabra[i] == 'o' ||
           palabra[i] == 'u' || palabra[i] == 'A' ||
           palabra[i] == 'E' || palabra[i] == 'I' ||
           palabra[i] == 'O' || palabra[i] == 'U') {
            vocales++;
        }

        // Contar vocal 'a' (incluye mayúscula)
        if(palabra[i] == 'a' || palabra[i] == 'A') {
            vocalesA++;
        }
        // Contar vocal 'e' (incluye mayúscula)
        else if(palabra[i] == 'e' || palabra[i] == 'E') {
            vocalesE++;
        }
        // Contar vocal 'i' (incluye mayúscula)
        else if(palabra[i] == 'i' || palabra[i] == 'I') {
            vocalesI++;
        }
        // Contar vocal 'o' (incluye mayúscula)
        else if(palabra[i] == 'o' || palabra[i] == 'O') {
            vocalesO++;
        }
        // Contar vocal 'u' (incluye mayúscula)
        else if(palabra[i] == 'u' || palabra[i] == 'U') {
            vocalesU++;
        }
    }

    // -------------------- RESULTADO PRINCIPAL --------------------
    cout << "La palabra " << palabra << " tiene " << vocales << " vocales." << endl << endl;

    // -------------------- RESUMEN POR VOCAL (SOLO LAS QUE APARECEN) --------------------
    cout << "RESUMEN DE LAS VOCALES" << endl;
    if(vocalesA > 0) {
        cout << "a = " << vocalesA << " veces." << endl;
    }
    if(vocalesE > 0) {
        cout << "e = " << vocalesE << " veces." << endl;
    }
    if(vocalesI > 0) {
        cout << "i = " << vocalesI << " veces." << endl;
    }
    if(vocalesO > 0) {
        cout << "o = " << vocalesO << " veces." << endl;
    }
    if(vocalesU > 0) {
        cout << "u = " << vocalesU << " veces." << endl;
    }

    // -------------------- DETERMINAR LA VOCAL (O VOCALES) MÁS REPETIDA --------------------
    // Se calcula el valor máximo entre los contadores individuales
    int peso = vocalesA;   // empezamos con el contador de 'a'

    if(vocalesE > peso) {
        peso = vocalesE;
    }
    if(vocalesI > peso) {
        peso = vocalesI;
    }
    if(vocalesO > peso) {
        peso = vocalesO;
    }
    if(vocalesU > peso) {
        peso = vocalesU;
    }

    // Se muestran todas las vocales cuyo contador sea igual al máximo
    cout << "La(s) vocal(es) mas repetida(s): ";
    if(vocalesA == peso) {
        cout << "a ";
    }
    if(vocalesE == peso) {
        cout << "e ";
    }
    if(vocalesI == peso) {
        cout << "i ";
    }
    if(vocalesO == peso) {
        cout << "o ";
    }
    if(vocalesU == peso) {
        cout << "u ";
    }
    cout << "con " << peso << " apariciones." << endl;

    return 0;
}
