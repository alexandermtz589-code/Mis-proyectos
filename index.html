#01 · Operaciones con 3 enteros suma | resta | producto
📌 Pide tres números enteros y muestra: suma, diferencia (a-b-c), producto y cociente (a/b/c). 
#include <iostream>
using namespace std;

int main()
{
    int a,b,c;

    cout<<"Dame un entero:"<<endl;
    cin>>a;

    cout<<"Dame otro entero:"<<endl;
    cin>>b;

    cout<<"Dame otro entero:"<<endl;
    cin>>c;

    cout<<"RESULTADOS"<<endl;
    cout<<"Suma: "<<a+b+c<<endl;
    cout<<"Diferencia: "<<a-b-c<<endl;
    cout<<"Producto: "<<a*b*c<<endl;
    cout<<"Cociente: "<<a/b/c<<endl;

    return 0;
}
#02 · Contador de vocales strings | bucle for
📌 Pide una palabra y cuenta cuántas vocales (a,e,i,o,u) contiene. Ejemplo: "computadora" → 5 vocales. 
/* Contar vocales en una palabra */

#include <iostream> 
using namespace std;

int main() {
    string palabra;
    int contador = 0;
    
    cout<<"Ingresa la palabra: ";
    cin>>palabra;
    
    for(int i = 0; i< palabra.length(); i++){
        if(palabra[i] == 'a'||palabra[i] == 'e'||
           palabra[i] == 'i'||palabra[i] == 'o'||
           palabra[i] == 'u')
        {
           contador++;
        }
    }
    
    cout<<"La palabra "<<palabra<<endl;
    cout<<"Vocales que tiene es:"<<contador;

    return 0;
}
